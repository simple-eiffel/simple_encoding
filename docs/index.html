simple_encoding

Character Encoding Library for Eiffel

GitHub Simple Eiffel

Overview

SIMPLE_ENCODING provides character encoding conversion for Eiffel applications:

  • UTF-8 <-> UTF-32 - Bidirectional conversion with error handling
  • Single-byte codecs - ISO-8859-1, ISO-8859-15, Windows-1252
  • Encoding detection - BOM detection and heuristics
  • Character properties - Unicode classification

Quick Start

local
    enc: SIMPLE_ENCODING
    utf8: STRING_8
do
    create enc.make

    -- Encode Unicode to UTF-8
    utf8 := enc.utf_32_to_utf_8 ("Hello World")

    -- Decode UTF-8 to Unicode
    text := enc.utf_8_to_utf_32 (utf8)

    -- Check for errors
    if enc.has_error then
        print (enc.last_error)
    end
end

Single-Byte Codecs

local
    registry: SIMPLE_CODEC_REGISTRY
    codec: detachable SIMPLE_CODEC
do
    create registry.make

    -- Lookup by name (supports aliases)
    codec := registry.codec_by_name ("latin1")
    codec := registry.codec_by_name ("cp1252")

    if attached codec as c then
        text := c.decode_string (bytes)
    end
end
